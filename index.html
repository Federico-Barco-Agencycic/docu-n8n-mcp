üìö Quick Start Examples

üîç Search for Slack nodes

// Fetch the search index
const response = await fetch('./api/search-index.json');
const nodes = await response.json();

// Find all Slack-related nodes
const slackNodes = nodes.filter(node => 
  node.keywords.some(keyword => keyword.includes('slack'))
);

console.log('Slack nodes:', slackNodes);

ü§ñ Get AI tools information

// Get all AI-capable nodes

const aiTools = await aiToolsResponse.json();

// Get detailed AI capabilities
const aiDetailsResponse = await fetch('./api/ai-tools-detailed.json');
const aiDetails = await aiDetailsResponse.json();

console.log(`Found ${aiTools.length} AI tools`);
